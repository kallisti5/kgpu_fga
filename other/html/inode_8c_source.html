781 static int truncate_upper(struct dentry *dentry, struct iattr *ia,
782  struct iattr *lower_ia)
783 {
784  int rc = 0;
785  struct inode *inode = dentry->d_inode;
786  struct ecryptfs_crypt_stat *crypt_stat;
787  loff_t i_size = i_size_read(inode);
788  loff_t lower_size_before_truncate;
789  loff_t lower_size_after_truncate;
790 
791  if (unlikely((ia->ia_size == i_size))) {
792  lower_ia->ia_valid &= ~ATTR_SIZE;
793  goto out;
794  }
795  crypt_stat = &ecryptfs_inode_to_private(dentry->d_inode)->crypt_stat;
796  /* Switch on growing or shrinking file */
797  if (ia->ia_size > i_size) {
798  char zero[] = { 0x00 };
799 
800  lower_ia->ia_valid &= ~ATTR_SIZE;
801  /* Write a single 0 at the last position of the file;
802  * this triggers code that will fill in 0's throughout
803  * the intermediate portion of the previous end of the
804  * file and the new and of the file */
805  rc = ecryptfs_write(inode, zero,
806  (ia->ia_size - 1), 1);
807  } else { /* ia->ia_size < i_size_read(inode) */
808  /* We're chopping off all the pages down to the page
809  * in which ia->ia_size is located. Fill in the end of
810  * that page from (ia->ia_size & ~PAGE_CACHE_MASK) to
811  * PAGE_CACHE_SIZE with zeros. */
812  size_t num_zeros = (PAGE_CACHE_SIZE
813  - (ia->ia_size & ~PAGE_CACHE_MASK));
814 
815 
816  /*
817  * XXX(truncate) this should really happen at the begginning
818  * of ->setattr. But the code is too messy to that as part
819  * of a larger patch. ecryptfs is also totally missing out
820  * on the inode_change_ok check at the beginning of
821  * ->setattr while would include this.
822  */
823  rc = inode_newsize_ok(inode, ia->ia_size);
824  if (rc)
825  goto out;
826 
827  if (!(crypt_stat->flags & ECRYPTFS_ENCRYPTED)) {
828  truncate_setsize(inode, ia->ia_size);
829  lower_ia->ia_size = ia->ia_size;
830  lower_ia->ia_valid |= ATTR_SIZE;
831  goto out;
832  }
833  if (num_zeros) {
834  char *zeros_virt;
835 
836  zeros_virt = kzalloc(num_zeros, GFP_KERNEL);
837  if (!zeros_virt) {
838  rc = -ENOMEM;
839  goto out;
840  }
841  rc = ecryptfs_write(inode, zeros_virt,
842  ia->ia_size, num_zeros);
843  kfree(zeros_virt);
844  if (rc) {
845  printk(KERN_ERR "Error attempting to zero out "
846  "the remainder of the end page on "
847  "reducing truncate; rc = [%d]\n", rc);
848  goto out;
849  }
850  }
851  truncate_setsize(inode, ia->ia_size);
852  rc = ecryptfs_write_inode_size_to_metadata(inode);
853  if (rc) {
854  printk(KERN_ERR "Problem with "
855  "ecryptfs_write_inode_size_to_metadata; "
856  "rc = [%d]\n", rc);
857  goto out;
858  }
859  /* We are reducing the size of the ecryptfs file, and need to
860  * know if we need to reduce the size of the lower file. */
861  lower_size_before_truncate =
862  upper_size_to_lower_size(crypt_stat, i_size);
863  lower_size_after_truncate =
864  upper_size_to_lower_size(crypt_stat, ia->ia_size);
865  if (lower_size_after_truncate < lower_size_before_truncate) {
866  lower_ia->ia_size = lower_size_after_truncate;
867  lower_ia->ia_valid |= ATTR_SIZE;
868  } else
869  lower_ia->ia_valid &= ~ATTR_SIZE;
870  }
871 out:
872  return rc;
873 }

922 static int ecryptfs_setattr(struct dentry *dentry, struct iattr *ia)
923 {
924  int rc = 0;
925  struct dentry *lower_dentry;
926  struct iattr lower_ia;
927  struct inode *inode;
928  struct inode *lower_inode;
929  struct ecryptfs_crypt_stat *crypt_stat;
930 
931  crypt_stat = &ecryptfs_inode_to_private(dentry->d_inode)->crypt_stat;
932  if (!(crypt_stat->flags & ECRYPTFS_STRUCT_INITIALIZED))
933  ecryptfs_init_crypt_stat(crypt_stat);
934  inode = dentry->d_inode;
935  lower_inode = ecryptfs_inode_to_lower(inode);
936  lower_dentry = ecryptfs_dentry_to_lower(dentry);
937  mutex_lock(&crypt_stat->cs_mutex);
938  if (S_ISDIR(dentry->d_inode->i_mode))
939  crypt_stat->flags &= ~(ECRYPTFS_ENCRYPTED);
940  else if (S_ISREG(dentry->d_inode->i_mode)
941  && (!(crypt_stat->flags & ECRYPTFS_POLICY_APPLIED)
942  || !(crypt_stat->flags & ECRYPTFS_KEY_VALID))) {
943  struct ecryptfs_mount_crypt_stat *mount_crypt_stat;
944 
945  mount_crypt_stat = &ecryptfs_superblock_to_private(
946  dentry->d_sb)->mount_crypt_stat;
947  rc = ecryptfs_read_metadata(dentry);
948  if (rc) {
949  if (!(mount_crypt_stat->flags
950  & ECRYPTFS_PLAINTEXT_PASSTHROUGH_ENABLED)) {
951  rc = -EIO;
952  printk(KERN_WARNING "Either the lower file "
953  "is not in a valid eCryptfs format, "
954  "or the key could not be retrieved. "
955  "Plaintext passthrough mode is not "
956  "enabled; returning -EIO\n");
957  mutex_unlock(&crypt_stat->cs_mutex);
958  goto out;
959  }
960  rc = 0;
961  crypt_stat->flags &= ~(ECRYPTFS_ENCRYPTED);
962  }
963  }
964  mutex_unlock(&crypt_stat->cs_mutex);
965  memcpy(&lower_ia, ia, sizeof(lower_ia));
966  if (ia->ia_valid & ATTR_FILE)
967  lower_ia.ia_file = ecryptfs_file_to_lower(ia->ia_file);
968  if (ia->ia_valid & ATTR_SIZE) {
969  rc = truncate_upper(dentry, ia, &lower_ia);
970  if (rc < 0)
971  goto out;
972  }
973 
974  /*
975  * mode change is for clearing setuid/setgid bits. Allow lower fs
976  * to interpret this in its own way.
977  */
978  if (lower_ia.ia_valid & (ATTR_KILL_SUID | ATTR_KILL_SGID))
979  lower_ia.ia_valid &= ~ATTR_MODE;
980 
981  mutex_lock(&lower_dentry->d_inode->i_mutex);
982  rc = notify_change(lower_dentry, &lower_ia);
983  mutex_unlock(&lower_dentry->d_inode->i_mutex);
984 out:
985  fsstack_copy_attr_all(inode, lower_inode);
986  return rc;
987 }
