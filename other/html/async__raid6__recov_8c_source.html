35 static struct dma_async_tx_descriptor *
36 async_sum_product(struct page *dest, struct page **srcs, unsigned char *coef,
37  size_t len, struct async_submit_ctl *submit)
38 {
39  struct dma_chan *chan = async_tx_find_channel(submit, DMA_PQ,
40  &dest, 1, srcs, 2, len);
41  struct dma_device *dma = chan ? chan->device : NULL;
42  const u8 *amul, *bmul;
43  u8 ax, bx;
44  u8 *a, *b, *c;
45 
46  if (dma) {
47  dma_addr_t dma_dest[2];
48  dma_addr_t dma_src[2];
49  struct device *dev = dma->dev;
50  struct dma_async_tx_descriptor *tx;
51  enum dma_ctrl_flags dma_flags = DMA_PREP_PQ_DISABLE_P;
52 
53  if (submit->flags & ASYNC_TX_FENCE)
54  dma_flags |= DMA_PREP_FENCE;
55  dma_dest[1] = dma_map_page(dev, dest, 0, len, DMA_BIDIRECTIONAL);
56  dma_src[0] = dma_map_page(dev, srcs[0], 0, len, DMA_TO_DEVICE);
57  dma_src[1] = dma_map_page(dev, srcs[1], 0, len, DMA_TO_DEVICE);
58  tx = dma->device_prep_dma_pq(chan, dma_dest, dma_src, 2, coef,
59  len, dma_flags);
60  if (tx) {
61  async_tx_submit(chan, tx, submit);
62  return tx;
63  }
64 
65  /* could not get a descriptor, unmap and fall through to
66  * the synchronous path
67  */
68  dma_unmap_page(dev, dma_dest[1], len, DMA_BIDIRECTIONAL);
69  dma_unmap_page(dev, dma_src[0], len, DMA_TO_DEVICE);
70  dma_unmap_page(dev, dma_src[1], len, DMA_TO_DEVICE);
71  }
72 
73  /* run the operation synchronously */
74  async_tx_quiesce(&submit->depend_tx);
75  amul = raid6_gfmul[coef[0]];
76  bmul = raid6_gfmul[coef[1]];
77  a = page_address(srcs[0]);
78  b = page_address(srcs[1]);
79  c = page_address(dest);
80 
81  while (len--) {
82  ax = amul[*a++];
83  bx = bmul[*b++];
84  *c++ = ax ^ bx;
85  }
86 
87  return NULL;
88 }

91 async_mult(struct page *dest, struct page *src, u8 coef, size_t len,
92  struct async_submit_ctl *submit)

97  const u8 *qmul; /* Q multiplier table */

100  if (dma) {
101  dma_addr_t dma_dest[2];
102  dma_addr_t dma_src[1];
103  struct device *dev = dma->dev;
104  struct dma_async_tx_descriptor *tx;
105  enum dma_ctrl_flags dma_flags = DMA_PREP_PQ_DISABLE_P;
106 
107  if (submit->flags & ASYNC_TX_FENCE)
108  dma_flags |= DMA_PREP_FENCE;
109  dma_dest[1] = dma_map_page(dev, dest, 0, len, DMA_BIDIRECTIONAL);
110  dma_src[0] = dma_map_page(dev, src, 0, len, DMA_TO_DEVICE);
111  tx = dma->device_prep_dma_pq(chan, dma_dest, dma_src, 1, &coef,
112  len, dma_flags);
113  if (tx) {
114  async_tx_submit(chan, tx, submit);
115  return tx;
116  }
117 
118  /* could not get a descriptor, unmap and fall through to
119  * the synchronous path
120  */
121  dma_unmap_page(dev, dma_dest[1], len, DMA_BIDIRECTIONAL);
122  dma_unmap_page(dev, dma_src[0], len, DMA_TO_DEVICE);
123  }
124 
125  /* no channel available, or failed to allocate a descriptor, so
126  * perform the operation synchronously
127  */
128  async_tx_quiesce(&submit->depend_tx);
129  qmul = raid6_gfmul[coef];
130  d = page_address(dest);
131  s = page_address(src);
132 
133  while (len--)
134  *d++ = qmul[*s++];
135 
136  return NULL;
137 }

139 static struct dma_async_tx_descriptor *
140 __2data_recov_4(int disks, size_t bytes, int faila, int failb,
141  struct page **blocks, struct async_submit_ctl *submit)
142 {
143  struct dma_async_tx_descriptor *tx = NULL;
144  struct page *p, *q, *a, *b;
145  struct page *srcs[2];
146  unsigned char coef[2];
147  enum async_tx_flags flags = submit->flags;
148  dma_async_tx_callback cb_fn = submit->cb_fn;
149  void *cb_param = submit->cb_param;
150  void *scribble = submit->scribble;
151 
152  p = blocks[disks-2];
153  q = blocks[disks-1];
154 
155  a = blocks[faila];
156  b = blocks[failb];
157 
158  /* in the 4 disk case P + Pxy == P and Q + Qxy == Q */
159  /* Dx = A*(P+Pxy) + B*(Q+Qxy) */
160  srcs[0] = p;
161  srcs[1] = q;
162  coef[0] = raid6_gfexi[failb-faila];
163  coef[1] = raid6_gfinv[raid6_gfexp[faila]^raid6_gfexp[failb]];
164  init_async_submit(submit, ASYNC_TX_FENCE, tx, NULL, NULL, scribble);
165  tx = async_sum_product(b, srcs, coef, bytes, submit);
166 
167  /* Dy = P+Pxy+Dx */
168  srcs[0] = p;
169  srcs[1] = b;
170  init_async_submit(submit, flags | ASYNC_TX_XOR_ZERO_DST, tx, cb_fn,
171  cb_param, scribble);
172  tx = async_xor(a, srcs, 0, 2, bytes, submit);
173 
174  return tx;
175 
176 }

417 static int sim_gpu_call(struct kgpu_request *r)
418 {
419  struct r62_recov_data *data = (struct r62_recov_data*)r->udata;
420  u8 *p, *q, *dp, *dq;
421  u8 px, qx;
422  const u8 *pbmul;
423  const u8 *qmul;
424 
425  int i, j;
426 
427  p = (u8*)r->in;
428  q = p+data->n*PAGE_SIZE;
429  dp = (u8*)r->out;
430  dq = dp+data->n*PAGE_SIZE;
431 
432  for (i=0; i<data->n; i++) {
433  pbmul = raid6_gfmul[data->idx[i].pbidx];
434  qmul = raid6_gfmul[data->idx[i].qidx];
435  for (j=0; j<PAGE_SIZE; j++) {
436  int id = j + i*PAGE_SIZE;
437  px = p[id] ^ dp[id];
438  qx = qmul[q[id] ^ dq[id]];
439  dq[id] = pbmul[px] ^ qx;
440  dp[id] = dq[id] ^ px;
441  }
442  }
443 
444  return 0;
445 }

447 static void process_r62_requests(struct list_head *reqs, int n, size_t tsz)
448 {
449  u8 *p, *q, *dp, *dq;
450  struct list_head *pos, *tmp;
451  struct r62_request *r = NULL;
452  int i, j;
453 
454  struct kgpu_request *greq = NULL;
455  u8 *gbuf = NULL;
456  struct r62_recov_data *data = NULL;
457 
458  size_t rsz = n<<(PAGE_SHIFT+1);
459  size_t dsz = sizeof(struct r62_recov_data)+n*sizeof(struct r62_tbl);
460 
461  //do_log(KGPU_LOG_PRINT, "#req = %d\n", n);
462 
463  gbuf = (u8*)kgpu_vmalloc(2*rsz+round_up(dsz, PAGE_SIZE));
464 
465  if (!gbuf) {
466  do_log(KGPU_LOG_ERROR, "out of GPU mem\n");
467  goto fail_out;
468  }
469 
470  greq = kgpu_alloc_request();
471  if (!greq) {
472  do_log(KGPU_LOG_ERROR, "can't alloc GPU request\n");
473  goto fail_out;
474  }
475 
476  greq->in = gbuf;
477  greq->out = gbuf + rsz + round_up(dsz, PAGE_SIZE);
478  greq->insize = 2*rsz+round_up(dsz, PAGE_SIZE);
479  greq->outsize = rsz;
480  greq->udata = gbuf + rsz;
481  greq->udatasize = dsz;
482 
483  data = (struct r62_recov_data*)greq->udata;
484  data->bytes = (size_t)(n*PAGE_SIZE);
485  data->n = n;
486 
487  p = (u8*)greq->in;
488  q = p + (n*PAGE_SIZE);
489  dp = (u8*)greq->out;
490  dq = dp + (n*PAGE_SIZE);
491 
492  j = 0;
493  list_for_each_safe(pos, tmp, reqs) {
494  u8 *tp, *tq, *tdp, *tdq;
495  r = list_entry(pos, struct r62_request, list);
496 
497  tp = r->blocks[r->disks-2];
498  tq = r->blocks[r->disks-1];
499 
500  tdp = r->blocks[r->faila];
501  r->blocks[r->faila] = (void *)raid6_empty_zero_page;
502  r->blocks[r->disks-2] = tdp;
503  tdq = r->blocks[r->failb];
504  r->blocks[r->failb] = (void *)raid6_empty_zero_page;
505  r->blocks[r->disks-1] = tdq;
506 
507  for (i=0; i<r->disks; i++) {
508  if (!r->blocks[i])
509  break;
510  }
511 
512  if (i== r->disks)
513  raid6_call.gen_syndrome(r->disks, r->bytes, (void**)r->blocks);
514  else {
515  prnt("NULL pointer at req %lu %d blk\n", r->id, i);
516  }
517 
518  r->status = R62_REQUEST_SYNGEN;
519 
520  r->blocks[r->faila] = tdp;
521  r->blocks[r->failb] = tdq;
522  r->blocks[r->disks-2] = tp;
523  r->blocks[r->disks-1] = tq;
524 
525  data->idx[j].pbidx = raid6_gfexi[r->failb - r->faila];
526  data->idx[j].qidx = raid6_gfinv[raid6_gfexp[r->faila]^raid6_gfexp[r->failb]];
527 
528  memcpy(p+(j<<PAGE_SHIFT),
529  tp,//page_address(r->blocks[r->disks-2]),
530  PAGE_SIZE);
531  memcpy(q+(j<<PAGE_SHIFT),
532  tq,//page_address(r->blocks[r->disks-1]),
533  PAGE_SIZE);
534  memcpy(dp+(j<<PAGE_SHIFT),
535  tdp,//page_address(r->blocks[r->faila]),
536  PAGE_SIZE);
537  memcpy(dq+(j<<PAGE_SHIFT),
538  tdq,//page_address(r->blocks[r->failb]),
539  PAGE_SIZE);
540 
541  prnt("handle req %lu %d\n", r->id, j);
542 
543  j++;
544  }
545 
546  strcpy(greq->service_name, "r62_recov");
547 
548  prnt("submit GPU request %d\n", greq->id);
549 
550  if (use_sim? sim_gpu_call(greq):kgpu_call_sync(greq)) {
551  do_log(KGPU_LOG_ERROR, "call gpu failed\n");
552  goto fail_out;
553  } else {
554  prnt("done GPU request %d\n", greq->id);
555  j = 0;
556  list_for_each_safe(pos, tmp, reqs) {
557  r = list_entry(pos, struct r62_request, list);
558 
559  memcpy(r->blocks[r->faila],
560  dp+(j<<PAGE_SHIFT), PAGE_SIZE);
561  memcpy(r->blocks[r->failb],
562  dq+(j<<PAGE_SHIFT), PAGE_SIZE);
563 
564  list_del(pos);
565 
566  r->status = R62_REQUEST_DONE;
567  prnt("to complete req %lu %d\n", r->id, j);
568  if (!no_merge)
569  complete(&r->c);
570  j++;
571  }
572  goto free_out;
573  }
574 
575 fail_out:
576  list_for_each_safe(pos, tmp, reqs) {
577  r = list_entry(pos, struct r62_request, list);
578  list_del(pos);
579  r->status = R62_REQUEST_ERROR;
580  if (likely(!no_merge))
581  complete(&r->c);
582  prnt("done req %lu with error\n", r->id);
583  }
584 
585 free_out:
586  if (gbuf) kgpu_vfree(gbuf);
587  if (greq) kgpu_free_request(greq);
588 }

742

748  * available punt to the synchronous path. In the 'dma not
749  * available' case be sure to use the scribble buffer to
750  * preserve the content of 'blocks' as the caller intended.
751  */
752  if (!async_dma_find_channel(DMA_PQ) || !scribble) {
753 
754  async_tx_quiesce(&submit->depend_tx);
755

758  // bufs. so we don't need re-aclloc new one.

760  u8 **ptrs = (u8**)blocks;

763  if (blocks[i] == NULL)
764  ptrs[i] = (u8 *) raid6_empty_zero_page;
765  else
766  ptrs[i] = (u8*)page_address(blocks[i]);

778

780  prnt("non-page size %lu\n", bytes);
781  }
782

784  if (blocks[i] == NULL)
785  ptrs[i] = (void *) raid6_empty_zero_page;
786  else
787  ptrs[i] = page_address(blocks[i]);

790  }

793 
794  return NULL;
795  }
796 
797  non_zero_srcs = 0;
798  for (i = 0; i < disks-2 && non_zero_srcs < 4; i++)
799  if (blocks[i])
800  non_zero_srcs++;
801  switch (non_zero_srcs) {
802  case 0:
803  case 1:
804  /* There must be at least 2 sources - the failed devices. */
805  BUG();
806 
807  case 2:

809  * operation (in contrast to the synchronous case), so
810  * explicitly handle the special case of a 4 disk array with

814  case 3:

816  * source pq operation (in contrast to the synchronous
817  * case), so explicitly handle the special case of a 5 disk
818  * array with 2 of 3 data disks missing.
819  */
820  return __2data_recov_5(disks, bytes, faila, failb, blocks, submit);
821  default:
822  return __2data_recov_n(disks, bytes, faila, failb, blocks, submit);

965 async_raid6_datap_recov(int disks, size_t bytes, int faila,
966  struct page **blocks, struct async_submit_ctl *submit)
967 {
968  struct dma_async_tx_descriptor *tx = NULL;
969  struct page *p, *q, *dq;
970  u8 coef;
971  enum async_tx_flags flags = submit->flags;
972  dma_async_tx_callback cb_fn = submit->cb_fn;
973  void *cb_param = submit->cb_param;
974  void *scribble = submit->scribble;
975  int good_srcs, good, i;
976  struct page *srcs[2];
977 
978  /* if a dma resource is not available or a scribble buffer is not
979  * available punt to the synchronous path. In the 'dma not
980  * available' case be sure to use the scribble buffer to
981  * preserve the content of 'blocks' as the caller intended.
982  */
983  if (!async_dma_find_channel(DMA_PQ) || !scribble) {
984  void **ptrs = scribble ? scribble : (void **) blocks;
985 
986  async_tx_quiesce(&submit->depend_tx);
987  for (i = 0; i < disks; i++)
988  if (blocks[i] == NULL)
989  ptrs[i] = (void*)raid6_empty_zero_page;
990  else
991  ptrs[i] = page_address(blocks[i]);
992 
993  raid6_datap_recov(disks, bytes, faila, ptrs);
994 
995  async_tx_sync_epilog(submit);
996 
997  return NULL;
998  }
999 
1000  good_srcs = 0;
1001  good = -1;
1002  for (i = 0; i < disks-2; i++) {
1003  if (i == faila)
1004  continue;
1005  if (blocks[i]) {
1006  good = i;
1007  good_srcs++;
1008  if (good_srcs > 1)
1009  break;
1010  }
1011  }
1012  BUG_ON(good_srcs == 0);
1013 
1014  p = blocks[disks-2];
1015  q = blocks[disks-1];
1016 
1017  /* Compute syndrome with zero for the missing data page
1018  * Use the dead data page as temporary storage for delta q
1019  */
1020  dq = blocks[faila];
1021  blocks[faila] = NULL;
1022  blocks[disks-1] = dq;
1023 
1024  /* in the 4-disk case we only need to perform a single source
1025  * multiplication with the one good data block.
1026  */
1027  if (good_srcs == 1) {
1028  struct page *g = blocks[good];
1029 
1030  init_async_submit(submit, ASYNC_TX_FENCE, tx, NULL, NULL,
1031  scribble);
1032  tx = async_memcpy(p, g, 0, 0, bytes, submit);
1033 
1034  init_async_submit(submit, ASYNC_TX_FENCE, tx, NULL, NULL,
1035  scribble);
1036  tx = async_mult(dq, g, raid6_gfexp[good], bytes, submit);
1037  } else {
1038  init_async_submit(submit, ASYNC_TX_FENCE, tx, NULL, NULL,
1039  scribble);
1040  tx = async_gen_syndrome(blocks, 0, disks, bytes, submit);
1041  }
1042 
1043  /* Restore pointer table */
1044  blocks[faila] = dq;
1045  blocks[disks-1] = q;
1046 
1047  /* calculate g^{-faila} */
1048  coef = raid6_gfinv[raid6_gfexp[faila]];
1049 
1050  srcs[0] = dq;
1051  srcs[1] = q;
1052  init_async_submit(submit, ASYNC_TX_FENCE|ASYNC_TX_XOR_DROP_DST, tx,
1053  NULL, NULL, scribble);
1054  tx = async_xor(dq, srcs, 0, 2, bytes, submit);
1055 
1056  init_async_submit(submit, ASYNC_TX_FENCE, tx, NULL, NULL, scribble);
1057  tx = async_mult(dq, dq, coef, bytes, submit);
1058 
1059  srcs[0] = p;
1060  srcs[1] = dq;
1061  init_async_submit(submit, flags | ASYNC_TX_XOR_DROP_DST, tx, cb_fn,
1062  cb_param, scribble);
1063  tx = async_xor(p, srcs, 0, 2, bytes, submit);
1064 
1065  return tx;
1066 }
