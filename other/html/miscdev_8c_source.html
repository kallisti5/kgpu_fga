241 ecryptfs_miscdev_read(struct file *file, char __user *buf, size_t count,
242  loff_t *ppos)
243 {
244  struct ecryptfs_daemon *daemon;
245  struct ecryptfs_msg_ctx *msg_ctx;
246  size_t packet_length_size;
247  char packet_length[3];
248  size_t i;
249  size_t total_length;
250  uid_t euid = current_euid();
251  int rc;
252 
253  mutex_lock(&ecryptfs_daemon_hash_mux);
254  /* TODO: Just use file->private_data? */
255  rc = ecryptfs_find_daemon_by_euid(&daemon, euid, current_user_ns());
256  BUG_ON(rc || !daemon);
257  mutex_lock(&daemon->mux);
258  if (daemon->flags & ECRYPTFS_DAEMON_ZOMBIE) {
259  rc = 0;
260  mutex_unlock(&ecryptfs_daemon_hash_mux);
261  printk(KERN_WARNING "%s: Attempt to read from zombified "
262  "daemon\n", __func__);
263  goto out_unlock_daemon;
264  }
265  if (daemon->flags & ECRYPTFS_DAEMON_IN_READ) {
266  rc = 0;
267  mutex_unlock(&ecryptfs_daemon_hash_mux);
268  goto out_unlock_daemon;
269  }
270  /* This daemon will not go away so long as this flag is set */
271  daemon->flags |= ECRYPTFS_DAEMON_IN_READ;
272  mutex_unlock(&ecryptfs_daemon_hash_mux);
273 check_list:
274  if (list_empty(&daemon->msg_ctx_out_queue)) {
275  mutex_unlock(&daemon->mux);
276  rc = wait_event_interruptible(
277  daemon->wait, !list_empty(&daemon->msg_ctx_out_queue));
278  mutex_lock(&daemon->mux);
279  if (rc < 0) {
280  rc = 0;
281  goto out_unlock_daemon;
282  }
283  }
284  if (daemon->flags & ECRYPTFS_DAEMON_ZOMBIE) {
285  rc = 0;
286  goto out_unlock_daemon;
287  }
288  if (list_empty(&daemon->msg_ctx_out_queue)) {
289  /* Something else jumped in since the
290  * wait_event_interruptable() and removed the
291  * message from the queue; try again */
292  goto check_list;
293  }
294  BUG_ON(euid != daemon->euid);
295  BUG_ON(current_user_ns() != daemon->user_ns);
296  BUG_ON(task_pid(current) != daemon->pid);
297  msg_ctx = list_first_entry(&daemon->msg_ctx_out_queue,
298  struct ecryptfs_msg_ctx, daemon_out_list);
299  BUG_ON(!msg_ctx);
300  mutex_lock(&msg_ctx->mux);
301  if (msg_ctx->msg) {
302  rc = ecryptfs_write_packet_length(packet_length,
303  msg_ctx->msg_size,
304  &packet_length_size);
305  if (rc) {
306  rc = 0;
307  printk(KERN_WARNING "%s: Error writing packet length; "
308  "rc = [%d]\n", __func__, rc);
309  goto out_unlock_msg_ctx;
310  }
311  } else {
312  packet_length_size = 0;
313  msg_ctx->msg_size = 0;
314  }
315  /* miscdevfs packet format:
316  * Octet 0: Type
317  * Octets 1-4: network byte order msg_ctx->counter
318  * Octets 5-N0: Size of struct ecryptfs_message to follow
319  * Octets N0-N1: struct ecryptfs_message (including data)
320  *
321  * Octets 5-N1 not written if the packet type does not
322  * include a message */
323  total_length = (1 + 4 + packet_length_size + msg_ctx->msg_size);
324  if (count < total_length) {
325  rc = 0;
326  printk(KERN_WARNING "%s: Only given user buffer of "
327  "size [%zd], but we need [%zd] to read the "
328  "pending message\n", __func__, count, total_length);
329  goto out_unlock_msg_ctx;
330  }
331  rc = -EFAULT;
332  if (put_user(msg_ctx->type, buf))
333  goto out_unlock_msg_ctx;
334  if (put_user(cpu_to_be32(msg_ctx->counter), (__be32 __user *)(buf + 1)))
335  goto out_unlock_msg_ctx;
336  i = 5;
337  if (msg_ctx->msg) {
338  if (copy_to_user(&buf[i], packet_length, packet_length_size))
339  goto out_unlock_msg_ctx;
340  i += packet_length_size;
341  if (copy_to_user(&buf[i], msg_ctx->msg, msg_ctx->msg_size))
342  goto out_unlock_msg_ctx;
343  i += msg_ctx->msg_size;
344  }
345  rc = i;
346  list_del(&msg_ctx->daemon_out_list);
347  kfree(msg_ctx->msg);
348  msg_ctx->msg = NULL;
349  /* We do not expect a reply from the userspace daemon for any
350  * message type other than ECRYPTFS_MSG_REQUEST */
351  if (msg_ctx->type != ECRYPTFS_MSG_REQUEST)
352  ecryptfs_msg_ctx_alloc_to_free(msg_ctx);
353 out_unlock_msg_ctx:
354  mutex_unlock(&msg_ctx->mux);
355 out_unlock_daemon:
356  daemon->flags &= ~ECRYPTFS_DAEMON_IN_READ;
357  mutex_unlock(&daemon->mux);
358  return rc;
359 }

409 static ssize_t
410 ecryptfs_miscdev_write(struct file *file, const char __user *buf,
411  size_t count, loff_t *ppos)
412 {
413  __be32 counter_nbo;
414  u32 seq;
415  size_t packet_size, packet_size_length, i;
416  ssize_t sz = 0;
417  char *data;
418  uid_t euid = current_euid();
419  int rc;
420 
421  if (count == 0)
422  goto out;
423 
424  data = memdup_user(buf, count);
425  if (IS_ERR(data)) {
426  printk(KERN_ERR "%s: memdup_user returned error [%ld]\n",
427  __func__, PTR_ERR(data));
428  goto out;
429  }
430  sz = count;
431  i = 0;
432  switch (data[i++]) {
433  case ECRYPTFS_MSG_RESPONSE:
434  if (count < (1 + 4 + 1 + sizeof(struct ecryptfs_message))) {
435  printk(KERN_WARNING "%s: Minimum acceptable packet "
436  "size is [%zd], but amount of data written is "
437  "only [%zd]. Discarding response packet.\n",
438  __func__,
439  (1 + 4 + 1 + sizeof(struct ecryptfs_message)),
440  count);
441  goto out_free;
442  }
443  memcpy(&counter_nbo, &data[i], 4);
444  seq = be32_to_cpu(counter_nbo);
445  i += 4;
446  rc = ecryptfs_parse_packet_length(&data[i], &packet_size,
447  &packet_size_length);
448  if (rc) {
449  printk(KERN_WARNING "%s: Error parsing packet length; "
450  "rc = [%d]\n", __func__, rc);
451  goto out_free;
452  }
453  i += packet_size_length;
454  if ((1 + 4 + packet_size_length + packet_size) != count) {
455  printk(KERN_WARNING "%s: (1 + packet_size_length([%zd])"
456  " + packet_size([%zd]))([%zd]) != "
457  "count([%zd]). Invalid packet format.\n",
458  __func__, packet_size_length, packet_size,
459  (1 + packet_size_length + packet_size), count);
460  goto out_free;
461  }
462  rc = ecryptfs_miscdev_response(&data[i], packet_size,
463  euid, current_user_ns(),
464  task_pid(current), seq);
465  if (rc)
466  printk(KERN_WARNING "%s: Failed to deliver miscdev "
467  "response to requesting operation; rc = [%d]\n",
468  __func__, rc);
469  break;
470  case ECRYPTFS_MSG_HELO:
471  case ECRYPTFS_MSG_QUIT:
472  break;
473  default:
474  ecryptfs_printk(KERN_WARNING, "Dropping miscdev "
475  "message of unrecognized type [%d]\n",
476  data[0]);
477  break;
478  }
479 out_free:
480  kfree(data);
481 out:
482  return sz;
483 }
