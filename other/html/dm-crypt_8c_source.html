206 /*
207  * Different IV generation algorithms:
208  *
209  * plain: the initial vector is the 32-bit little-endian version of the sector
210  * number, padded with zeros if necessary.
211  *
212  * plain64: the initial vector is the 64-bit little-endian version of the sector
213  * number, padded with zeros if necessary.
214  *
215  * essiv: "encrypted sector|salt initial vector", the sector number is
216  * encrypted with the bulk cipher using a salt as key. The salt
217  * should be derived from the bulk cipher's key via hashing.
218  *
219  * benbi: the 64-bit "big-endian 'narrow block'-count", starting at 1
220  * (needed for LRW-32-AES and possible other narrow block modes)
221  *
222  * null: the initial vector is always zero. Provides compatibility with
223  * obsolete loop_fish2 devices. Do not use for new devices.
224  *
225  * lmk: Compatible implementation of the block chaining mode used
226  * by the Loop-AES block device encryption system
227  * designed by Jari Ruusu. See http://loop-aes.sourceforge.net/
228  * It operates on full 512 byte sectors and uses CBC
229  * with an IV derived from the sector number, the data and
230  * optionally extra IV seed.
231  * This means that after decryption the first block
232  * of sector must be tweaked according to decrypted data.
233  * Loop-AES can use three encryption schemes:
234  * version 1: is plain aes-cbc mode
235  * version 2: uses 64 multikey scheme with lmk IV generator
236  * version 3: the same as version 2 with additional IV seed
237  * (it uses 65 keys, last key is used as IV seed)
238  *
239  * plumb: unimplemented, see:
240  * http://article.gmane.org/gmane.linux.kernel.device-mapper.dm-crypt/454
241  */

367 static int crypt_iv_essiv_ctr(struct crypt_config *cc, struct dm_target *ti,
368  const char *opts)
369 {
370  struct crypto_cipher *essiv_tfm = NULL;
371  struct crypto_hash *hash_tfm = NULL;
372  u8 *salt = NULL;
373  int err, cpu;
374 
375  if (!opts) {
376  ti->error = "Digest algorithm missing for ESSIV mode";
377  return -EINVAL;
378  }
379 
380  /* Allocate hash algorithm */
381  hash_tfm = crypto_alloc_hash(opts, 0, CRYPTO_ALG_ASYNC);
382  if (IS_ERR(hash_tfm)) {
383  ti->error = "Error initializing ESSIV hash";
384  err = PTR_ERR(hash_tfm);
385  goto bad;
386  }
387 
388  salt = kzalloc(crypto_hash_digestsize(hash_tfm), GFP_KERNEL);
389  if (!salt) {
390  ti->error = "Error kmallocing salt storage in ESSIV";
391  err = -ENOMEM;
392  goto bad;
393  }
394 
395  cc->iv_gen_private.essiv.salt = salt;
396  cc->iv_gen_private.essiv.hash_tfm = hash_tfm;
397 
398  for_each_possible_cpu(cpu) {
399  essiv_tfm = setup_essiv_cpu(cc, ti, salt,
400  crypto_hash_digestsize(hash_tfm));
401  if (IS_ERR(essiv_tfm)) {
402  crypt_iv_essiv_dtr(cc);
403  return PTR_ERR(essiv_tfm);
404  }
405  per_cpu_ptr(cc->cpu, cpu)->iv_private = essiv_tfm;
406  }
407 
408  return 0;
409 
410 bad:
411  if (hash_tfm && !IS_ERR(hash_tfm))
412  crypto_free_hash(hash_tfm);
413  kfree(salt);
414  return err;
415 }

429 static int crypt_iv_benbi_ctr(struct crypt_config *cc, struct dm_target *ti,
430  const char *opts)
431 {
432  unsigned bs = crypto_ablkcipher_blocksize(any_tfm(cc));
433  int log = ilog2(bs);
434 
435  /* we need to calculate how far we must shift the sector count
436  * to get the cipher block count, we use this shift in _gen */
437 
438  if (1 << log != bs) {
439  ti->error = "cypher blocksize is not a power of 2";
440  return -EINVAL;
441  }
442 
443  if (log > 9) {
444  ti->error = "cypher blocksize is > 512";
445  return -EINVAL;
446  }
447 
448  cc->iv_gen_private.benbi.shift = 9 - log;
449 
450  return 0;
451 }

540 static int crypt_iv_lmk_one(struct crypt_config *cc, u8 *iv,
541  struct dm_crypt_request *dmreq,
542  u8 *data)
543 {
544  struct iv_lmk_private *lmk = &cc->iv_gen_private.lmk;
545  struct {
546  struct shash_desc desc;
547  char ctx[crypto_shash_descsize(lmk->hash_tfm)];
548  } sdesc;
549  struct md5_state md5state;
550  u32 buf[4];
551  int i, r;
552 
553  sdesc.desc.tfm = lmk->hash_tfm;
554  sdesc.desc.flags = CRYPTO_TFM_REQ_MAY_SLEEP;
555 
556  r = crypto_shash_init(&sdesc.desc);
557  if (r)
558  return r;
559 
560  if (lmk->seed) {
561  r = crypto_shash_update(&sdesc.desc, lmk->seed, LMK_SEED_SIZE);
562  if (r)
563  return r;
564  }
565 
566  /* Sector is always 512B, block size 16, add data of blocks 1-31 */
567  r = crypto_shash_update(&sdesc.desc, data + 16, 16 * 31);
568  if (r)
569  return r;
570 
571  /* Sector is cropped to 56 bits here */
572  buf[0] = cpu_to_le32(dmreq->iv_sector & 0xFFFFFFFF);
573  buf[1] = cpu_to_le32((((u64)dmreq->iv_sector >> 32) & 0x00FFFFFF) | 0x80000000);
574  buf[2] = cpu_to_le32(4024);
575  buf[3] = 0;
576  r = crypto_shash_update(&sdesc.desc, (u8 *)buf, sizeof(buf));
577  if (r)
578  return r;
579 
580  /* No MD5 padding here */
581  r = crypto_shash_export(&sdesc.desc, &md5state);
582  if (r)
583  return r;
584 
585  for (i = 0; i < MD5_HASH_WORDS; i++)
586  __cpu_to_le32s(&md5state.hash[i]);
587  memcpy(iv, &md5state.hash, cc->iv_size);
588 
589  return 0;
590 }

906 /*
907  * Generate a new unfragmented bio with the given size
908  * This should never violate the device limitations
909  * May return a smaller bio when running out of pages, indicated by
910  * *out_of_pages set to 1.
911  */
912 static struct bio *crypt_alloc_buffer(struct dm_crypt_io *io, unsigned size,
913  unsigned *out_of_pages)
914 {
915  struct crypt_config *cc = io->target->private;
916  struct bio *clone;
917  unsigned int nr_iovecs = (size + PAGE_SIZE - 1) >> PAGE_SHIFT;
918  gfp_t gfp_mask = GFP_NOIO | __GFP_HIGHMEM;
919  unsigned i, len;
920  struct page *page;
921 
922  clone = bio_alloc_bioset(GFP_NOIO, nr_iovecs, cc->bs);
923  if (!clone)
924  return NULL;
925 
926  clone_init(io, clone);
927  *out_of_pages = 0;
928 
929  for (i = 0; i < nr_iovecs; i++) {
930  page = mempool_alloc(cc->page_pool, gfp_mask);
931  if (!page) {
932  *out_of_pages = 1;
933  break;
934  }
935 
936  /*
937  * if additional pages cannot be allocated without waiting,
938  * return a partially allocated bio, the caller will then try
939  * to allocate additional bios while submitting this partial bio
940  */
941  if (i == (MIN_BIO_PAGES - 1))
942  gfp_mask = (gfp_mask | __GFP_NOWARN) & ~__GFP_WAIT;
943 
944  len = (size > PAGE_SIZE) ? PAGE_SIZE : size;
945 
946  if (!bio_add_page(clone, page, len, 0)) {
947  mempool_free(page, cc->page_pool);
948  break;
949  }
950 
951  size -= len;
952  }
953 
954  if (!clone->bi_size) {
955  bio_put(clone);
956  return NULL;
957  }
958 
959  return clone;
960 }
