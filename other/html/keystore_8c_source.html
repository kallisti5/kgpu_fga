1311  printk(KERN_ERR "First byte != 0x%.2x; invalid packet\n",
1312  ECRYPTFS_TAG_3_PACKET_TYPE);
1313  rc = -EINVAL;
1314  goto out;
1315  }
1316  /* Released: wipe_auth_tok_list called in ecryptfs_parse_packet_set or
1317  * at end of function upon failure */
1318  auth_tok_list_item =
1319  kmem_cache_zalloc(ecryptfs_auth_tok_list_item_cache, GFP_KERNEL);
1320  if (!auth_tok_list_item) {
1321  printk(KERN_ERR "Unable to allocate memory\n");
1322  rc = -ENOMEM;
1323  goto out;
1324  }
1325  (*new_auth_tok) = &auth_tok_list_item->auth_tok;
1326  rc = ecryptfs_parse_packet_length(&data[(*packet_size)], &body_size,
1327  &length_size);
1328  if (rc) {
1329  printk(KERN_WARNING "Error parsing packet length; rc = [%d]\n",
1330  rc);
1331  goto out_free;
1332  }
1333  if (unlikely(body_size < (ECRYPTFS_SALT_SIZE + 5))) {
1334  printk(KERN_WARNING "Invalid body size ([%td])\n", body_size);
1335  rc = -EINVAL;
1336  goto out_free;
1337  }
1338  (*packet_size) += length_size;
1339  if (unlikely((*packet_size) + body_size > max_packet_size)) {
1340  printk(KERN_ERR "Packet size exceeds max\n");
1341  rc = -EINVAL;
1342  goto out_free;
1343  }
1344  (*new_auth_tok)->session_key.encrypted_key_size =
1345  (body_size - (ECRYPTFS_SALT_SIZE + 5));
1346  if ((*new_auth_tok)->session_key.encrypted_key_size
1347  > ECRYPTFS_MAX_ENCRYPTED_KEY_BYTES) {
1348  printk(KERN_WARNING "Tag 3 packet contains key larger "
1349  "than ECRYPTFS_MAX_ENCRYPTED_KEY_BYTES\n");
1350  rc = -EINVAL;
1351  goto out_free;
1352  }
1353  if (unlikely(data[(*packet_size)++] != 0x04)) {
1354  printk(KERN_WARNING "Unknown version number [%d]\n",
1355  data[(*packet_size) - 1]);
1356  rc = -EINVAL;
1357  goto out_free;
1358  }
1359  rc = ecryptfs_cipher_code_to_string(crypt_stat->cipher,
1360  (u16)data[(*packet_size)]);
1361  if (rc)
1362  goto out_free;
1363  /* A little extra work to differentiate among the AES key
1364  * sizes; see RFC2440 */
1365  switch(data[(*packet_size)++]) {
1366  case RFC2440_CIPHER_AES_192:
1367  crypt_stat->key_size = 24;
1368  break;
1369  default:
1370  crypt_stat->key_size =
1371  (*new_auth_tok)->session_key.encrypted_key_size;
1372  }
1373  rc = ecryptfs_init_crypt_ctx(crypt_stat);
1374  if (rc)
1375  goto out_free;
1376  if (unlikely(data[(*packet_size)++] != 0x03)) {
1377  printk(KERN_WARNING "Only S2K ID 3 is currently supported\n");
1378  rc = -ENOSYS;
1379  goto out_free;
1380  }
1381  /* TODO: finish the hash mapping */
1382  switch (data[(*packet_size)++]) {
1383  case 0x01: /* See RFC2440 for these numbers and their mappings */
1384  /* Choose MD5 */
1385  memcpy((*new_auth_tok)->token.password.salt,
1386  &data[(*packet_size)], ECRYPTFS_SALT_SIZE);
1387  (*packet_size) += ECRYPTFS_SALT_SIZE;
1388  /* This conversion was taken straight from RFC2440 */
1389  (*new_auth_tok)->token.password.hash_iterations =
1390  ((u32) 16 + (data[(*packet_size)] & 15))
1391  << ((data[(*packet_size)] >> 4) + 6);
1392  (*packet_size)++;
1393  /* Friendly reminder:
1394  * (*new_auth_tok)->session_key.encrypted_key_size =
1395  * (body_size - (ECRYPTFS_SALT_SIZE + 5)); */
1396  memcpy((*new_auth_tok)->session_key.encrypted_key,
1397  &data[(*packet_size)],
1398  (*new_auth_tok)->session_key.encrypted_key_size);
1399  (*packet_size) +=
1400  (*new_auth_tok)->session_key.encrypted_key_size;
1401  (*new_auth_tok)->session_key.flags &=
1402  ~ECRYPTFS_CONTAINS_DECRYPTED_KEY;
1403  (*new_auth_tok)->session_key.flags |=
1404  ECRYPTFS_CONTAINS_ENCRYPTED_KEY;
1405  (*new_auth_tok)->token.password.hash_algo = 0x01; /* MD5 */
1406  break;
1407  default:
1408  ecryptfs_printk(KERN_ERR, "Unsupported hash algorithm: "
1409  "[%d]\n", data[(*packet_size) - 1]);
1410  rc = -ENOSYS;
1411  goto out_free;
1412  }
1413  (*new_auth_tok)->token_type = ECRYPTFS_PASSWORD;
1414  /* TODO: Parametarize; we might actually want userspace to
1415  * decrypt the session key. */
1416  (*new_auth_tok)->session_key.flags &=
1417  ~(ECRYPTFS_USERSPACE_SHOULD_TRY_TO_DECRYPT);
1418  (*new_auth_tok)->session_key.flags &=
1419  ~(ECRYPTFS_USERSPACE_SHOULD_TRY_TO_ENCRYPT);
1420  list_add(&auth_tok_list_item->list, auth_tok_list);
1421  goto out;
1422 out_free:
1423  (*new_auth_tok) = NULL;
1424  memset(auth_tok_list_item, 0,
1425  sizeof(struct ecryptfs_auth_tok_list_item));
1426  kmem_cache_free(ecryptfs_auth_tok_list_item_cache,
1427  auth_tok_list_item);
1428 out:
1429  if (rc)
1430  (*packet_size) = 0;
1431  return rc;
1432 }

1450 static int
1451 parse_tag_11_packet(unsigned char *data, unsigned char *contents,
1452  size_t max_contents_bytes, size_t *tag_11_contents_size,
1453  size_t *packet_size, size_t max_packet_size)
1454 {
1455  size_t body_size;
1456  size_t length_size;
1457  int rc = 0;
1458 
1459  (*packet_size) = 0;
1460  (*tag_11_contents_size) = 0;
1461  /* This format is inspired by OpenPGP; see RFC 2440
1462  * packet tag 11
1463  *
1464  * Tag 11 identifier (1 byte)
1465  * Max Tag 11 packet size (max 3 bytes)
1466  * Binary format specifier (1 byte)
1467  * Filename length (1 byte)
1468  * Filename ("_CONSOLE") (8 bytes)
1469  * Modification date (4 bytes)
1470  * Literal data (arbitrary)
1471  *
1472  * We need at least 16 bytes of data for the packet to even be
1473  * valid.
1474  */
1475  if (max_packet_size < 16) {
1476  printk(KERN_ERR "Maximum packet size too small\n");
1477  rc = -EINVAL;
1478  goto out;
1479  }
1480  if (data[(*packet_size)++] != ECRYPTFS_TAG_11_PACKET_TYPE) {
1481  printk(KERN_WARNING "Invalid tag 11 packet format\n");
1482  rc = -EINVAL;
1483  goto out;
1484  }
1485  rc = ecryptfs_parse_packet_length(&data[(*packet_size)], &body_size,
1486  &length_size);
1487  if (rc) {
1488  printk(KERN_WARNING "Invalid tag 11 packet format\n");
1489  goto out;
1490  }
1491  if (body_size < 14) {
1492  printk(KERN_WARNING "Invalid body size ([%td])\n", body_size);
1493  rc = -EINVAL;
1494  goto out;
1495  }
1496  (*packet_size) += length_size;
1497  (*tag_11_contents_size) = (body_size - 14);
1498  if (unlikely((*packet_size) + body_size + 1 > max_packet_size)) {
1499  printk(KERN_ERR "Packet size exceeds max\n");
1500  rc = -EINVAL;
1501  goto out;
1502  }
1503  if (unlikely((*tag_11_contents_size) > max_contents_bytes)) {
1504  printk(KERN_ERR "Literal data section in tag 11 packet exceeds "
1505  "expected size\n");
1506  rc = -EINVAL;
1507  goto out;
1508  }
1509  if (data[(*packet_size)++] != 0x62) {
1510  printk(KERN_WARNING "Unrecognizable packet\n");
1511  rc = -EINVAL;
1512  goto out;
1513  }
1514  if (data[(*packet_size)++] != 0x08) {
1515  printk(KERN_WARNING "Unrecognizable packet\n");
1516  rc = -EINVAL;
1517  goto out;
1518  }
1519  (*packet_size) += 12; /* Ignore filename and modification date */
1520  memcpy(contents, &data[(*packet_size)], (*tag_11_contents_size));
1521  (*packet_size) += (*tag_11_contents_size);
1522 out:
1523  if (rc) {
1524  (*packet_size) = 0;
1525  (*tag_11_contents_size) = 0;
1526  }
1527  return rc;
1528 }

1610 static int
1611 decrypt_passphrase_encrypted_session_key(struct ecryptfs_auth_tok *auth_tok,
1612  struct ecryptfs_crypt_stat *crypt_stat)
1613 {
1614  struct scatterlist dst_sg[2];
1615  struct scatterlist src_sg[2];
1616  struct mutex *tfm_mutex;
1617  struct blkcipher_desc desc = {
1618  .flags = CRYPTO_TFM_REQ_MAY_SLEEP
1619  };
1620  int rc = 0;
1621 
1622  if (unlikely(ecryptfs_verbosity > 0)) {
1623  ecryptfs_printk(
1624  KERN_DEBUG, "Session key encryption key (size [%d]):\n",
1625  auth_tok->token.password.session_key_encryption_key_bytes);
1626  ecryptfs_dump_hex(
1627  auth_tok->token.password.session_key_encryption_key,
1628  auth_tok->token.password.session_key_encryption_key_bytes);
1629  }
1630  rc = ecryptfs_get_tfm_and_mutex_for_cipher_name(&desc.tfm, &tfm_mutex,
1631  crypt_stat->cipher);
1632  if (unlikely(rc)) {
1633  printk(KERN_ERR "Internal error whilst attempting to get "
1634  "tfm and mutex for cipher name [%s]; rc = [%d]\n",
1635  crypt_stat->cipher, rc);
1636  goto out;
1637  }
1638  rc = virt_to_scatterlist(auth_tok->session_key.encrypted_key,
1639  auth_tok->session_key.encrypted_key_size,
1640  src_sg, 2);
1641  if (rc < 1 || rc > 2) {
1642  printk(KERN_ERR "Internal error whilst attempting to convert "
1643  "auth_tok->session_key.encrypted_key to scatterlist; "
1644  "expected rc = 1; got rc = [%d]. "
1645  "auth_tok->session_key.encrypted_key_size = [%d]\n", rc,
1646  auth_tok->session_key.encrypted_key_size);
1647  goto out;
1648  }
1649  auth_tok->session_key.decrypted_key_size =
1650  auth_tok->session_key.encrypted_key_size;
1651  rc = virt_to_scatterlist(auth_tok->session_key.decrypted_key,
1652  auth_tok->session_key.decrypted_key_size,
1653  dst_sg, 2);
1654  if (rc < 1 || rc > 2) {
1655  printk(KERN_ERR "Internal error whilst attempting to convert "
1656  "auth_tok->session_key.decrypted_key to scatterlist; "
1657  "expected rc = 1; got rc = [%d]\n", rc);
1658  goto out;
1659  }
1660  mutex_lock(tfm_mutex);
1661  rc = crypto_blkcipher_setkey(
1662  desc.tfm, auth_tok->token.password.session_key_encryption_key,
1663  crypt_stat->key_size);
1664  if (unlikely(rc < 0)) {
1665  mutex_unlock(tfm_mutex);
1666  printk(KERN_ERR "Error setting key for crypto context\n");
1667  rc = -EINVAL;
1668  goto out;
1669  }
1670  rc = crypto_blkcipher_decrypt(&desc, dst_sg, src_sg,
1671  auth_tok->session_key.encrypted_key_size);
1672  mutex_unlock(tfm_mutex);
1673  if (unlikely(rc)) {
1674  printk(KERN_ERR "Error decrypting; rc = [%d]\n", rc);
1675  goto out;
1676  }
1677  auth_tok->session_key.flags |= ECRYPTFS_CONTAINS_DECRYPTED_KEY;
1678  memcpy(crypt_stat->key, auth_tok->session_key.decrypted_key,
1679  auth_tok->session_key.decrypted_key_size);
1680  crypt_stat->flags |= ECRYPTFS_KEY_VALID;
1681  if (unlikely(ecryptfs_verbosity > 0)) {
1682  ecryptfs_printk(KERN_DEBUG, "FEK of size [%zd]:\n",
1683  crypt_stat->key_size);
1684  ecryptfs_dump_hex(crypt_stat->key,
1685  crypt_stat->key_size);
1686  }
1687 out:
1688  return rc;
1689 }

1704 int ecryptfs_parse_packet_set(struct ecryptfs_crypt_stat *crypt_stat,
1705  unsigned char *src,
1706  struct dentry *ecryptfs_dentry)
1707 {
1708  size_t i = 0;
1709  size_t found_auth_tok;
1710  size_t next_packet_is_auth_tok_packet;
1711  struct list_head auth_tok_list;
1712  struct ecryptfs_auth_tok *matching_auth_tok;
1713  struct ecryptfs_auth_tok *candidate_auth_tok;
1714  char *candidate_auth_tok_sig;
1715  size_t packet_size;
1716  struct ecryptfs_auth_tok *new_auth_tok;
1717  unsigned char sig_tmp_space[ECRYPTFS_SIG_SIZE];
1718  struct ecryptfs_auth_tok_list_item *auth_tok_list_item;
1719  size_t tag_11_contents_size;
1720  size_t tag_11_packet_size;
1721  struct key *auth_tok_key = NULL;
1722  int rc = 0;
1723 
1724  INIT_LIST_HEAD(&auth_tok_list);
1725  /* Parse the header to find as many packets as we can; these will be
1726  * added the our &auth_tok_list */
1727  next_packet_is_auth_tok_packet = 1;
1728  while (next_packet_is_auth_tok_packet) {
1729  size_t max_packet_size = ((PAGE_CACHE_SIZE - 8) - i);
1730 
1731  switch (src[i]) {
1732  case ECRYPTFS_TAG_3_PACKET_TYPE:
1733  rc = parse_tag_3_packet(crypt_stat,
1734  (unsigned char *)&src[i],
1735  &auth_tok_list, &new_auth_tok,
1736  &packet_size, max_packet_size);
1737  if (rc) {
1738  ecryptfs_printk(KERN_ERR, "Error parsing "
1739  "tag 3 packet\n");
1740  rc = -EIO;
1741  goto out_wipe_list;
1742  }
1743  i += packet_size;
1744  rc = parse_tag_11_packet((unsigned char *)&src[i],
1745  sig_tmp_space,
1746  ECRYPTFS_SIG_SIZE,
1747  &tag_11_contents_size,
1748  &tag_11_packet_size,
1749  max_packet_size);
1750  if (rc) {
1751  ecryptfs_printk(KERN_ERR, "No valid "
1752  "(ecryptfs-specific) literal "
1753  "packet containing "
1754  "authentication token "
1755  "signature found after "
1756  "tag 3 packet\n");
1757  rc = -EIO;
1758  goto out_wipe_list;
1759  }
1760  i += tag_11_packet_size;
1761  if (ECRYPTFS_SIG_SIZE != tag_11_contents_size) {
1762  ecryptfs_printk(KERN_ERR, "Expected "
1763  "signature of size [%d]; "
1764  "read size [%zd]\n",
1765  ECRYPTFS_SIG_SIZE,
1766  tag_11_contents_size);
1767  rc = -EIO;
1768  goto out_wipe_list;
1769  }
1770  ecryptfs_to_hex(new_auth_tok->token.password.signature,
1771  sig_tmp_space, tag_11_contents_size);
1772  new_auth_tok->token.password.signature[
1773  ECRYPTFS_PASSWORD_SIG_SIZE] = '\0';
1774  crypt_stat->flags |= ECRYPTFS_ENCRYPTED;
1775  break;
1776  case ECRYPTFS_TAG_1_PACKET_TYPE:
1777  rc = parse_tag_1_packet(crypt_stat,
1778  (unsigned char *)&src[i],
1779  &auth_tok_list, &new_auth_tok,
1780  &packet_size, max_packet_size);
1781  if (rc) {
1782  ecryptfs_printk(KERN_ERR, "Error parsing "
1783  "tag 1 packet\n");
1784  rc = -EIO;
1785  goto out_wipe_list;
1786  }
1787  i += packet_size;
1788  crypt_stat->flags |= ECRYPTFS_ENCRYPTED;
1789  break;
1790  case ECRYPTFS_TAG_11_PACKET_TYPE:
1791  ecryptfs_printk(KERN_WARNING, "Invalid packet set "
1792  "(Tag 11 not allowed by itself)\n");
1793  rc = -EIO;
1794  goto out_wipe_list;
1795  break;
1796  default:
1797  ecryptfs_printk(KERN_DEBUG, "No packet at offset [%zd] "
1798  "of the file header; hex value of "
1799  "character is [0x%.2x]\n", i, src[i]);
1800  next_packet_is_auth_tok_packet = 0;
1801  }
1802  }
1803  if (list_empty(&auth_tok_list)) {
1804  printk(KERN_ERR "The lower file appears to be a non-encrypted "
1805  "eCryptfs file; this is not supported in this version "
1806  "of the eCryptfs kernel module\n");
1807  rc = -EINVAL;
1808  goto out;

1813  * necessary to decrypt the EFEK in the auth_tok parsed from
1814  * the metadata. There may be several potential matches, but
1815  * just one will be sufficient to decrypt to get the FEK. */
1816 find_next_matching_auth_tok:

1819  key_put(auth_tok_key);
1820  auth_tok_key = NULL;
1821  }

1824  if (unlikely(ecryptfs_verbosity > 0)) {
1825  ecryptfs_printk(KERN_DEBUG,
1826  "Considering cadidate auth tok:\n");
1827  ecryptfs_dump_auth_tok(candidate_auth_tok);
1828  }
1829  rc = ecryptfs_get_auth_tok_sig(&candidate_auth_tok_sig,
1830  candidate_auth_tok);
1831  if (rc) {
1832  printk(KERN_ERR
1833  "Unrecognized candidate auth tok type: [%d]\n",
1834  candidate_auth_tok->token_type);
1835  rc = -EINVAL;
1836  goto out_wipe_list;
1837  }

1839  &matching_auth_tok,
1840  crypt_stat->mount_crypt_stat,
1841  candidate_auth_tok_sig);
1842  if (!rc) {
1843  found_auth_tok = 1;
1844  goto found_matching_auth_tok;
1845  }
1846  }
1847  if (!found_auth_tok) {
1848  ecryptfs_printk(KERN_ERR, "Could not find a usable "
1849  "authentication token\n");
1850  rc = -EIO;
1851  goto out_wipe_list;
1852  }
1853 found_matching_auth_tok:
1854  if (candidate_auth_tok->token_type == ECRYPTFS_PRIVATE_KEY) {
1855  memcpy(&(candidate_auth_tok->token.private_key),
1856  &(matching_auth_tok->token.private_key),
1857  sizeof(struct ecryptfs_private_key));
1858  rc = decrypt_pki_encrypted_session_key(candidate_auth_tok,
1859  crypt_stat);
1860  } else if (candidate_auth_tok->token_type == ECRYPTFS_PASSWORD) {
1861  memcpy(&(candidate_auth_tok->token.password),
1862  &(matching_auth_tok->token.password),
1863  sizeof(struct ecryptfs_password));
1864  rc = decrypt_passphrase_encrypted_session_key(
1865  candidate_auth_tok, crypt_stat);
1866  }
1867  if (rc) {

1869 
1870  ecryptfs_printk(KERN_WARNING, "Error decrypting the "
1871  "session key for authentication token with sig "
1872  "[%.*s]; rc = [%d]. Removing auth tok "
1873  "candidate from the list and searching for "
1874  "the next match.\n", ECRYPTFS_SIG_SIZE_HEX,
1875  candidate_auth_tok_sig, rc);

1877  auth_tok_list_item_tmp,
1878  &auth_tok_list, list) {
1879  if (candidate_auth_tok
1880  == &auth_tok_list_item->auth_tok) {
1881  list_del(&auth_tok_list_item->list);
1882  kmem_cache_free(
1883  ecryptfs_auth_tok_list_item_cache,
1884  auth_tok_list_item);
1885  goto find_next_matching_auth_tok;
1886  }
1887  }
1888  BUG();
1889  }
1890  rc = ecryptfs_compute_root_iv(crypt_stat);
1891  if (rc) {
1892  ecryptfs_printk(KERN_ERR, "Error computing "
1893  "the root IV\n");
1894  goto out_wipe_list;
