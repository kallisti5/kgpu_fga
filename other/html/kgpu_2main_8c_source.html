158 
159  data = kmem_cache_alloc(kgpu_sync_call_data_cache, GFP_KERNEL);
160  if (!data) {
161  kgpu_log(KGPU_LOG_ERROR, "kgpu_call_sync alloc mem failed\n");
162  return -ENOMEM;
163  }
164  item = kmem_cache_alloc(kgpu_request_item_cache, GFP_KERNEL);
165  if (!item) {
166  kgpu_log(KGPU_LOG_ERROR, "out of memory for kgpu request\n");
167  return -ENOMEM;
168  }
169  item->r = req;
170 
171  data->oldkdata = req->kdata;
172  data->oldcallback = req->callback;
173  data->done = 0;
174  init_waitqueue_head(&data->queue);
175 
176  req->kdata = data;
177  req->callback = sync_callback;
178 
179  spin_lock(&(kgpudev.reqlock));
180 
181  INIT_LIST_HEAD(&item->list);
182  list_add_tail(&item->list, &(kgpudev.reqs));
183 
184  wake_up_interruptible(&(kgpudev.reqq));
185 
186  spin_unlock(&(kgpudev.reqlock));
187 
188  wait_event_interruptible(data->queue, (data->done==1));
189 
190  req->kdata = data->oldkdata;
191  req->callback = data->oldcallback;
192  kmem_cache_free(kgpu_sync_call_data_cache, data);
193 
194  return 0;

287 void kgpu_vfree(void *p)
288 {
289  unsigned long idx =
290  ((unsigned long)(p)
291  - (unsigned long)(kgpudev.gmpool.kva))/KGPU_BUF_UNIT_SIZE;
292  unsigned int nunits;
293 
294  if (idx < 0 || idx >= kgpudev.gmpool.nunits) {
295  kgpu_log(KGPU_LOG_ERROR,
296  "incorrect GPU memory pointer 0x%lX to free\n",
297  p);
298  return;
299  }
300 
301  nunits = kgpudev.gmpool.alloc_sz[idx];
302  if (nunits == 0) {
303  /*
304  * We allow such case because this allows users free memory
305  * from any field among in, out and data in request.
306  */
307  return;
308  }
309  if (nunits > (kgpudev.gmpool.nunits - idx)) {
310  kgpu_log(KGPU_LOG_ERROR, "incorrect GPU memory allocation info: "
311  "allocated %u units at unit index %u\n", nunits, idx);
312  return;
313  }
314 
315  spin_lock(&kgpudev.gmpool_lock);
316  bitmap_clear(kgpudev.gmpool.bitmap, idx, nunits);
317  kgpudev.gmpool.alloc_sz[idx] = 0;
318  spin_unlock(&kgpudev.gmpool_lock);
319 }

322 unsigned long kgpu_alloc_mmap_area(unsigned long size)
323 {
324  unsigned int n = DIV_ROUND_UP(size, PAGE_SIZE);
325  unsigned long p = 0;
326  unsigned long idx;
327 
328  spin_lock(&kgpudev.vm_lock);
329 
330  idx = bitmap_find_next_zero_area(
331  kgpudev.vm.bitmap,
332  kgpudev.vm.npages,
333  0, n, 0);
334  if (idx < kgpudev.vm.npages) {
335  bitmap_set(kgpudev.vm.bitmap, idx, n);
336  p = kgpudev.vm.start + PAGE_SIZE*idx;
337  kgpudev.vm.alloc_sz[idx] = n;
338  } else {
339  kgpu_log(KGPU_LOG_ERROR, "out of mmap area for mapping "
340  "ask for %u page %lu size, idx %lu\n",
341  n, size, idx);
342  }
343 
344  spin_unlock(&kgpudev.vm_lock);
345

442 static void* map_page_units(void *units, int n, int is_page)
443 {
444  unsigned long addr;
445  int i;
446  int ret;
447 
448  addr = kgpu_alloc_mmap_area(n<<PAGE_SHIFT);
449  if (!addr) {
450  return NULL;
451  }
452 
453  down_write(&kgpudev.vm.vma->vm_mm->mmap_sem);
454 
455  for (i=0; i<n; i++) {
456  ret = vm_insert_page(
457  kgpudev.vm.vma,
458  addr+i*PAGE_SIZE,
459  is_page? ((struct page**)units)[i] : pfn_to_page(((unsigned long*)units)[i])
460  );
461 
462  if (unlikely(ret < 0)) {
463  up_write(&kgpudev.vm.vma->vm_mm->mmap_sem);
464  kgpu_log(KGPU_LOG_ERROR,
465  "can't remap pfn %lu, error code %d\n",
466  is_page ? page_to_pfn(((struct page**)units)[i]) : ((unsigned long*)units)[i],
467  ret);
468  return NULL;
469  }
470  }
471 
472  up_write(&kgpudev.vm.vma->vm_mm->mmap_sem);
473 
474  return (void*)addr;
475 
476 }

571 ssize_t kgpu_read(
572  struct file *filp, char __user *buf, size_t c, loff_t *fpos)
573 {
574  ssize_t ret = 0;
575  struct list_head *r;
576  struct _kgpu_request_item *item;
577 
578  spin_lock(&(kgpudev.reqlock));
579  while (list_empty(&(kgpudev.reqs))) {
580  spin_unlock(&(kgpudev.reqlock));
581 
582  if (filp->f_flags & O_NONBLOCK)
583  return -EAGAIN;
584 
585  if (wait_event_interruptible(
586  kgpudev.reqq, (!list_empty(&(kgpudev.reqs)))))
587  return -ERESTARTSYS;
588  spin_lock(&(kgpudev.reqlock));
589  }
590 
591  r = kgpudev.reqs.next;
592  list_del(r);
593  item = list_entry(r, struct _kgpu_request_item, list);
594  if (item) {
595  struct kgpu_ku_request kureq;
596  fill_ku_request(&kureq, item->r);
597 
598  memcpy(buf, &kureq, sizeof(struct kgpu_ku_request));
599  ret = c;
600  }
601 
602  spin_unlock(&(kgpudev.reqlock));
603 
604  if (ret > 0 && item) {
605  spin_lock(&(kgpudev.rtdreqlock));
606 
607  INIT_LIST_HEAD(&item->list);
608  list_add_tail(&item->list, &(kgpudev.rtdreqs));
609 
610  spin_unlock(&(kgpudev.rtdreqlock));
611  }
612 
613  *fpos += ret;
614 
615  return ret;
616 }

618 ssize_t kgpu_write(struct file *filp, const char __user *buf,
619  size_t count, loff_t *fpos)
620 {
621  struct kgpu_ku_response kuresp;
622  struct _kgpu_request_item *item;
623  ssize_t ret = 0;
624  size_t realcount;
625 
626  if (count < sizeof(struct kgpu_ku_response))
627  ret = -EINVAL; /* Too small. */
628  else
629  {
630  realcount = sizeof(struct kgpu_ku_response);
631 
632  memcpy/*copy_from_user*/(&kuresp, buf, realcount);
633 
634  item = find_request(kuresp.id, 1);
635  if (!item)
636  {
637  ret = -EFAULT; /* no request found */
638  } else {
639  item->r->errcode = kuresp.errcode;
640  if (unlikely(kuresp.errcode != 0)) {
641  switch(kuresp.errcode) {
642  case KGPU_NO_RESPONSE:
643  kgpu_log(KGPU_LOG_ALERT,
644  "userspace helper doesn't give any response\n");
645  break;
646  case KGPU_NO_SERVICE:
647  kgpu_log(KGPU_LOG_ALERT,
648  "no such service %s\n",
649  item->r->service_name);
650  break;
651  case KGPU_TERMINATED:
652  kgpu_log(KGPU_LOG_ALERT,
653  "request is terminated\n"
654  );
655  break;
656  default:
657  kgpu_log(KGPU_LOG_ALERT,
658  "unknown error with code %d\n",
659  kuresp.id);
660  break;
661  }
662  }
663 
664  /*
665  * Different strategy should be applied here:
666  * #1 invoke the callback in the write syscall, like here.
667  * #2 add the resp into the resp-list in the write syscall
668  * and return, a kernel thread will process the list
669  * and invoke the callback.
670  *
671  * Currently, the first one is used because this can ensure
672  * the fast response. A kthread may have to sleep so that
673  * the response can't be processed ASAP.
674  */
675  item->r->callback(item->r);
676  ret = count;/*realcount;*/
677  *fpos += ret;
678  kmem_cache_free(kgpu_request_item_cache, item);
679  }
680  }

703 {
704  struct kgpu_gpu_mem_info gb;
705  struct _kgpu_mempool *gmp = &kgpudev.gmpool;
706  int i;
707  int err=0;
708 
709  spin_lock(&(kgpudev.gmpool_lock));

712

715  gmp->npages = gb.size/PAGE_SIZE;

731  gmp->nunits = gmp->npages/KGPU_BUF_NR_FRAMES_PER_UNIT;
732  if (!gmp->bitmap) {

741  bitmap_zero(gmp->bitmap, gmp->nunits);
742 
743  /* set up allocated memory sizes */
744  if (!gmp->alloc_sz) {
745  gmp->alloc_sz = kmalloc(
746  gmp->nunits*sizeof(u32), GFP_KERNEL);
747  if (!gmp->alloc_sz) {
748  kgpu_log(KGPU_LOG_ERROR,
749  "run out of memory for gmp alloc_sz\n");
750  err = -ENOMEM;

752  }
753  }

761  err = -EFAULT;
762  goto unlock_and_out;
763  }

787 {

790  if (_IOC_TYPE(cmd) != KGPU_IOC_MAGIC)

795  err = !access_ok(VERIFY_WRITE, (void __user *)arg, _IOC_SIZE(cmd));
796  else if (_IOC_DIR(cmd) & _IOC_WRITE)
797  err = !access_ok(VERIFY_READ, (void __user *)arg, _IOC_SIZE(cmd));

809 
810  case KGPU_IOC_SET_STOP:
811  err = terminate_all_requests();
812  break;
813 
814  default:
815  err = -ENOTTY;
816  break;
817  }
818 
819  return err;
